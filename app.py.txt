from flask import Flask, request, send_file, jsonify
import pandas as pd
import requests
import os

app = Flask(__name__)

GROQ_API_KEY = "PASTE_YOUR_GROQ_API_KEY_HERE"

@app.route("/")
def home():
    return "Excel AI API is running"

@app.route("/process", methods=["POST"])
def process_excel():
    if "file" not in request.files or "command" not in request.form:
        return jsonify({"error": "File or command missing"}), 400

    file = request.files["file"]
    command = request.form["command"]

    df = pd.read_excel(file)

    # Simple AI decision logic (safe MVP)
    cmd = command.lower()

    if "remove duplicate" in cmd:
        df = df.drop_duplicates()

    elif "sort" in cmd:
        df = df.sort_values(by=df.columns[0])

    elif "count" in cmd:
        return jsonify({"rows": len(df)})

    output_path = "output.xlsx"
    df.to_excel(output_path, index=False)

    return send_file(output_path, as_attachment=True)

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=3000)
